\doxysection{Programos veikimas}\label{md__r_e_a_d_m_e_autotoc_md1}

\begin{DoxyItemize}
\item Paleidus programą, vartotojo yra paklausiama ar jis nori pats irašyti duomenis ar visus veiksmus atlikti su failais
\item Pasirinkus rašyti rankiniu būdu, vartotojo yra prašoma įvesti studento vardą bei pavardę
\item Tada yra paklausiama ar yra žinomas pažymių skaičius, bei kokiu būdu norima įvesti tuos pažymius (įrašyti ranka arba generuoti atsitiktinai)
\item Įrašius visus pažymius (įskaitant ir egzamino vertinimą), vartotojui leidžiama pasirinkti ar jis nori pridėti dar vieną studenta
\item Įrašius norimą kiekį studentų, vartotojas turi galimybę pasirinkti rezultatus\+: vidurkį arba medianą
\end{DoxyItemize}

\DoxyHorRuler{0}



\begin{DoxyItemize}
\item Pasirinkus atlikti veiksmus su failais, vartotojo yra paklausiama ar jis nori generuoti duomenų failus.
\item Pasirinkus generuoti, vartotojui leidžiama rinktis kuriuos failus nori generuoti
\item Po failų sukūrimo, vartotojas turi pasirinkti kurį failą jis nori nuskaityti
\item Galiausiai vartotojui duodamas pasirinkimas pagal ką rūšiuoti studentus \+: vidurkį ar medianą
\item Tuomet rezultatai yra išvedami į 2 atskirus failus\+: vargšiukai ir kietiakai (kietiakai, kurių vidurkis $>$= 5)
\end{DoxyItemize}

\DoxyHorRuler{0}

\begin{DoxyItemize}
\item Paleidus programą su pasirinktu konteineriu (parametrai paleidimo metu), visi programos veiksmai bus atlikti automatiškai\+:
\begin{DoxyItemize}
\item Programa nustatys ar truksta duomenų failų -\/ jei kurio nors nebus, jį sugeneruos
\item Praėjus duomenų failų tikrinimą, programa nuskaitys duomenis į pasirinktą konteinerį(arba konteinerius, jei bus įvesti keli konteinerių tipai)
\item Nuskaitytus duomenis programa išrūšiuoja į kietiakus ir vargšiukus (vargšiukai -\/ studentai, kurių pažymiai $<$ 5)
\item Galiausiai visi rezultatai būna sulygiuojami ir išvedami į failus
\end{DoxyItemize}
\end{DoxyItemize}
